# select
# 	c_name,
# 	c_custkey,
# 	o_orderkey,
# 	o_orderdate,
# 	o_totalprice,
# 	sum(l_quantity)
# from
# 	customer,
# 	orders,
# 	lineitem
# where
# 	o_orderkey in (
# 		select
# 			l_orderkey
# 		from
# 			lineitem
# 		group by
# 			l_orderkey having
# 				sum(l_quantity) > :1
# 	)
# 	and c_custkey = o_custkey
# 	and o_orderkey = l_orderkey
# group by
# 	c_name,
# 	c_custkey,
# 	o_orderkey,
# 	o_orderdate,
# 	o_totalprice
# order by
# 	o_totalprice desc,
# 	o_orderdate;
# :n 100

orderby([o_totalprice, o_orderdate],
  groupby([c_name,
           c_custkey,
           o_orderkey,
           o_orderdate,
           o_totalprice,
           sum(l_quantity)],
    [c_name, c_custkey, o_orderkey, o_orderdate, o_totalprice],
    join((c_custkey) = (o_custkey),
      join((o_orderkey) = (l_orderkey),
        filter(exists(filter(((l_orderkey) = (o_orderkey)) && ((sum_l_quantity) > (param1)),
                        groupby([l_orderkey,
                                 sum(l_quantity) as sum_l_quantity],
                          [l_orderkey],
                          lineitem))),
          orders),
        lineitem),
      customer)),
  desc)
